function toWeirdCase(string){
  let res='';
  let index=0;
  for(let i=0;i<string.length;i++){
    let c=string[i];
    if(c==' '){
      res+=' ';
      index=0;
    }else{
    if(index%2==0){
      res+=c.toUpperCase();
    
    }
    else{
      res+=c.toLowerCase();
      
    }
      index++;
    }
  }
  return res;
}
