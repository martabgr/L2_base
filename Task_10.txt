function bingo(ticket, win){
  let w=0;
  for(let i=0;i<ticket.length;i++){
    let letters=ticket[i][0];
    let nums=ticket[i][1];
    
    for(let k=0;k<letters.length;k++){
      if(letters.charCodeAt(k)==nums){
        w++;break;
      }
    }
  }
  if(w>=win){
    return "Winner!";
  }
  else{
    return "Loser!";
  }
}
