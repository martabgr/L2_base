function hexStringToRGB(hexString) {
  let s = '';
    for (let i = 0; i < hexString.length; i++) {
        if (hexString[i] !== '#') {
            s += hexString[i];
        }
    }
    let r = parseInt(s.substring(0, 2), 16);
    let g = parseInt(s.substring(2, 4), 16);
    let b = parseInt(s.substring(4, 6), 16);

      return { r: r, g: g, b: b };
}
